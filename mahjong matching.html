    <script>

        let currentLevel = 1;

        let errorsThisRound = 0;

        let startTime;

        let timerInterval;

        let tilesRemaining = 0;

        let selectedTile = null;


        const tilePairs = [

            ['🀇', 'char'], ['🀈', 'char'], ['🀉', 'char'],

            ['🀄', 'honor'], ['🀆', 'honor'], ['🀀', 'honor'], 

            ['🀙', 'dot'], ['🀚', 'dot'], ['🀛', 'dot'],

            ['🀐', 'bamboo'], ['🀑', 'bamboo'], ['🀒', 'bamboo']

        ];


        const levels = {

            1: { rows: 4, cols: 4, maxTime: 45, maxErr: 2 },

            2: { rows: 4, cols: 6, maxTime: 70, maxErr: 3 },

            3: { rows: 6, cols: 6, maxTime: 110, maxErr: 4 },

            4: { rows: 6, cols: 8, maxTime: 150, maxErr: 5 }

        };


        function initGame() {

            document.getElementById('start-screen').style.display = 'none';

            document.getElementById('game-screen').style.display = 'flex';

            currentLevel = 1;

            startRound();

        }


        // --- NEW: Return to Home Function ---

        function returnToHome() {

            clearInterval(timerInterval); // Stop the clock

            document.getElementById('game-screen').style.display = 'none';

            document.getElementById('game-over').style.display = 'none';

            document.getElementById('start-screen').style.display = 'block';

        }


        function startRound() {

            document.getElementById('game-over').style.display = 'none';

            const board = document.getElementById('game-board');

            board.innerHTML = '';

            

            errorsThisRound = 0;

            selectedTile = null;

            document.getElementById('error-display').innerText = errorsThisRound;

            document.getElementById('level-display').innerText = currentLevel;

            

            const config = levels[currentLevel];

            board.style.gridTemplateColumns = `repeat(${config.cols}, 1fr)`;


            let fullSet = [];

            const pairsRequired = (config.rows * config.cols) / 2;

            for(let i=0; i < pairsRequired; i++) {

                let pairIdx = i % tilePairs.length; 

                fullSet.push(tilePairs[pairIdx]);

                fullSet.push(tilePairs[pairIdx]);

            }


            fullSet.sort(() => Math.random() - 0.5);

            tilesRemaining = fullSet.length;


            for(let r=0; r < config.rows; r++) {

                for(let c=0; c < config.cols; c++) {

                    const info = fullSet.pop();

                    createTile(r, c, info[0], info[1]);

                }

            }


            updateTilesStatus();

            

            let safety = 0;

            while (!checkAvailableMoves() && safety < 5) {

                executeSmartShuffle();

                safety++;

            }

            startTimer();

        }


        function createTile(row, col, symbol, suit) {

            const tile = document.createElement('div');

            tile.className = 'tile';

            tile.innerText = symbol;

            tile.id = `tile-${row}-${col}`;

            tile.dataset.row = row;

            tile.dataset.col = col;

            tile.dataset.suit = suit;

            tile.dataset.symbol = symbol;

            tile.addEventListener('click', handleTileClick);

            document.getElementById('game-board').appendChild(tile);

        }


        function handleTileClick(event) {

            const tile = event.target;

            if (tile.classList.contains('locked') || tile.classList.contains('matched')) return;


            if (selectedTile === null) {

                selectedTile = tile;

                tile.classList.add('selected');

            } else if (selectedTile === tile) {

                selectedTile.classList.remove('selected');

                selectedTile = null;

            } else {

                if (tile.dataset.symbol === selectedTile.dataset.symbol) {

                    matchTiles(selectedTile, tile);

                } else {

                    errorsThisRound++;

                    document.getElementById('error-display').innerText = errorsThisRound;

                    selectedTile.classList.remove('selected');

                }

                selectedTile = null;

            }

        }


        function matchTiles(t1, t2) {

            t1.classList.remove('selected');

            t1.classList.add('matched');

            t2.classList.add('matched');

            tilesRemaining -= 2;


            updateTilesStatus();


            if (tilesRemaining === 0) {

                endRound();

            } else if (!checkAvailableMoves()) {

                setTimeout(() => {

                    alert("🤖 AI 分析：檢測到死局！系統已自動重新洗牌。");

                    executeSmartShuffle();

                }, 300);

            }

        }


        function updateTilesStatus() {

            const tiles = document.querySelectorAll('.tile:not(.matched)');

            tiles.forEach(t => {

                const r = parseInt(t.dataset.row);

                const c = parseInt(t.dataset.col);

                const left = document.getElementById(`tile-${r}-${c-1}`);

                const right = document.getElementById(`tile-${r}-${c+1}`);


                const hasLeft = left && !left.classList.contains('matched');

                const hasRight = right && !right.classList.contains('matched');


                if (!hasLeft || !hasRight) {

                    t.classList.remove('locked');

                } else {

                    t.classList.add('locked');

                }

            });

        }


        function checkAvailableMoves() {

            const free = Array.from(document.querySelectorAll('.tile:not(.locked):not(.matched)'));

            for (let i = 0; i < free.length; i++) {

                for (let j = i + 1; j < free.length; j++) {

                    if (free[i].dataset.symbol === free[j].dataset.symbol) return true;

                }

            }

            return false;

        }


        function forceManualShuffle() {

            executeSmartShuffle();

        }


        function executeSmartShuffle() {

            const remaining = Array.from(document.querySelectorAll('.tile:not(.matched)'));

            const free = remaining.filter(t => !t.classList.contains('locked'));


            if (free.length >= 2) {

                let dataPool = remaining.map(t => ({ symbol: t.dataset.symbol, suit: t.dataset.suit }));

                

                let counts = {};

                let target = null;

                for (let d of dataPool) {

                    counts[d.symbol] = (counts[d.symbol] || 0) + 1;

                    if (counts[d.symbol] === 2) { target = d; break; }

                }


                if (target) {

                    let removed = 0;

                    dataPool = dataPool.filter(d => {

                        if (d.symbol === target.symbol && removed < 2) { removed++; return false; }

                        return true;

                    });

                    

                    dataPool.sort(() => Math.random() - 0.5);


                    free[0].dataset.symbol = target.symbol;

                    free[0].dataset.suit = target.suit;

                    free[0].innerText = target.symbol;


                    free[1].dataset.symbol = target.symbol;

                    free[1].dataset.suit = target.suit;

                    free[1].innerText = target.symbol;


                    let i = 0;

                    remaining.forEach(t => {

                        if (t !== free[0] && t !== free[1]) {

                            t.dataset.symbol = dataPool[i].symbol;

                            t.dataset.suit = dataPool[i].suit;

                            t.innerText = dataPool[i].symbol;

                            i++;

                        }

                        t.classList.remove('selected');

                        t.classList.add('shuffling');

                        setTimeout(() => t.classList.remove('shuffling'), 500);

                    });

                }

            }

            selectedTile = null;

            updateTilesStatus();

        }


        function startTimer() {

            clearInterval(timerInterval);

            startTime = Date.now();

            timerInterval = setInterval(() => {

                document.getElementById('time-display').innerText = Math.floor((Date.now() - startTime) / 1000);

            }, 1000);

        }


        function endRound() {

            clearInterval(timerInterval);

            const time = Math.floor((Date.now() - startTime) / 1000);

            const config = levels[currentLevel];

            let fb = `⏱️ 耗時: ${time} 秒<br>❌ 錯誤: ${errorsThisRound} 次<br><br>`;


            if (errorsThisRound === 0 && time < config.maxTime) {

                if (currentLevel < 4) {

                    currentLevel++;

                    fb += "🤖 AI 分析：表現優異！已升級難度。";

                } else {

                    fb += "🤖 AI 分析：完美通關！已達最高難度。";

                }

            } else if (errorsThisRound > config.maxErr || time > config.maxTime + 15) {

                if (currentLevel > 1) {

                    currentLevel--;

                    fb += "🤖 AI 分析：已為您降低難度，請放鬆心情。";

                } else {

                    fb += "🤖 AI 分析：請保持節奏，慢慢尋找。";

                }

            } else {

                fb += "🤖 AI 分析：表現穩定，維持當前難度。";

            }


            document.getElementById('ai-feedback').innerHTML = fb;

            document.getElementById('game-over').style.display = 'block';

        }

    </script>
